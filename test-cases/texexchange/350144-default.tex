			colorbar sampled,
			% define the style of the colorbar
			colorbar style={

					% ... with the anchor ...

					parent axis width=3*(0.28\textwidth-width("300")) + 2*20pt,
					% ... and the start and end points ...

					samples={
							\pgfplotscolormapsizeof{%
								\pgfkeysvalueof{/pgfplots/colormap name}%
							}+1
						},
				},
		]
